See below for quick copy-paste templates for adding GeckoLib to your development environment.

## Version reference
In your `gradle.properties` or `libs.versions.toml` file:

<Tabs groupId="versionschema" queryString="versionschema">
    <TabItem value="gradleproperties" label="gradle.properties">
        ```properties
        geckolibVersion=
        ```
    </TabItem>
    <TabItem value="versioncatalogue" label="libs.versions.toml">
        ```toml
        geckolib =
        ```
    </TabItem>
</Tabs>

## Repository declaration
In your `build.gradle` file:

<Tabs groupId="dsl" queryString="dsl">
    <TabItem value="groovy" label="Groovy">
        ```groovy
        repositories {
            exclusiveContent {
                forRepository {
                    maven {
                        name = 'GeckoLib'
                        url = 'https://dl.cloudsmith.io/public/geckolib3/geckolib/maven/'
                    }
                }
                filter { includeGroupAndSubgroups('software.bernie.geckolib') }
            }
        }
        ```
    </TabItem>
    <TabItem value="kotlin" label="Kotlin">
        ```kotlin
        repositories {
            exclusiveContent {
                forRepository {
                    maven {
                        name = "GeckoLib"
                        url = uri("https://dl.cloudsmith.io/public/geckolib3/geckolib/maven/")
                    }
                }
                filter { includeGroupAndSubgroups("software.bernie.geckolib") }
            }
        }
        ```
    </TabItem>
</Tabs>

## Dependency declaration
<Tabs groupId="versionschema" queryString="versionschema">
    <TabItem value="gradleproperties" label="gradle.properties">
        In your `build.gradle` file, inside the `dependencies` block:

        <Tabs groupId="dsl" queryString="dsl">
            <TabItem value="groovy" label="Groovy">
                ```groovy
                implementation "software.bernie.geckolib:geckolib-neoforge-${minecraftVersion}:${geckolibVersion}"
                ```
            </TabItem>
            <TabItem value="kotlin" label="Kotlin">
                ```kotlin
                implementation("software.bernie.geckolib:geckolib-neoforge-${minecraftVersion}:${geckolibVersion}")
                ```
            </TabItem>
        </Tabs>
    </TabItem>
    <TabItem value="versioncatalogue" label="libs.versions.toml">
        In your `libs.versions.toml`, under the `[libraries]` section:

        ```groovy
        geckolib = { group = "software.bernie.geckolib", name = "geckolib-neoforge-1.21.11", version.ref = "geckolib" }
        ```

        :::note
        Make sure you change `1.21.11` to your intended Minecraft version
        :::

        Then, in your `build.gradle` file:

        <Tabs groupId="dsl" queryString="dsl">
            <TabItem value="groovy" label="Groovy">
                ```groovy
                implementation libs.geckolib
                ```
            </TabItem>
            <TabItem value="kotlin" label="Kotlin">
                ```kotlin
                implementation(libs.geckolib)
                ```
            </TabItem>
        </Tabs>
    </TabItem>
</Tabs>
